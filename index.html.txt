    function setLang(lang) {
      const faItems = document.querySelectorAll('[id$="-fa"], #fa-interests');
      const enItems = document.querySelectorAll('[id$="-en"], #en-interests');

      if (lang === 'fa') {
        faItems.forEach(el => el.style.display = '');
        enItems.forEach(el => el.style.display = 'none');
        document.body.dir = 'rtl';
      } else {
        faItems.forEach(el => el.style.display = 'none');
        enItems.forEach(el => el.style.display = '';
        document.body.dir = 'ltr';
      }
    }
